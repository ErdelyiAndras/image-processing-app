option(ENABLE_LOGGING "Enable verbose logging output" OFF)


set(_KERNELS_DIR  "${CMAKE_CURRENT_SOURCE_DIR}/../kernels")
set(_GENERATED_DIR "${CMAKE_CURRENT_BINARY_DIR}/generated")
file(MAKE_DIRECTORY "${_GENERATED_DIR}")

set(_KERNEL_FILES
    "${_KERNELS_DIR}/utils.cl"
    "${_KERNELS_DIR}/denoising/TVDenoising.cl"
    "${_KERNELS_DIR}/denoising/GaussianBlur.cl"
    "${_KERNELS_DIR}/edge-detection/SobelEdgeDetection.cl"
    "${_KERNELS_DIR}/edge-detection/CannyEdgeDetection.cl"
    "${_KERNELS_DIR}/shape-detection/HoughLineShapeDetection.cl"
    "${_KERNELS_DIR}/shape-detection/HoughCircleShapeDetection.cl"
)

set_property(DIRECTORY APPEND PROPERTY CMAKE_CONFIGURE_DEPENDS ${_KERNEL_FILES})


function(_embed_kernel FILE CONSTANT_NAME OUT_VAR)
    file(READ "${FILE}" _SRC)
    string(APPEND ${OUT_VAR}
        "static constexpr const char* ${CONSTANT_NAME} = R\"CL(\n${_SRC}\n)CL\";\n\n"
    )
    set(${OUT_VAR} "${${OUT_VAR}}" PARENT_SCOPE)
endfunction()

set(_HEADER
    "// Auto-generated by CMake — do not edit.\n"
    "// Regenerated automatically whenever any .cl file changes.\n"
    "#pragma once\n\n"
)
string(CONCAT _HEADER ${_HEADER})

_embed_kernel("${_KERNELS_DIR}/utils.cl"
    "UTILS_KERNEL_SOURCE"                       _HEADER)
_embed_kernel("${_KERNELS_DIR}/denoising/TVDenoising.cl"
    "TV_DENOISING_KERNEL_SOURCE"                _HEADER)
_embed_kernel("${_KERNELS_DIR}/denoising/GaussianBlur.cl"
    "GAUSSIAN_BLUR_KERNEL_SOURCE"               _HEADER)
_embed_kernel("${_KERNELS_DIR}/edge-detection/SobelEdgeDetection.cl"
    "SOBEL_EDGE_DETECTION_KERNEL_SOURCE"        _HEADER)
_embed_kernel("${_KERNELS_DIR}/edge-detection/CannyEdgeDetection.cl"
    "CANNY_EDGE_DETECTION_KERNEL_SOURCE"        _HEADER)
_embed_kernel("${_KERNELS_DIR}/shape-detection/HoughLineShapeDetection.cl"
    "HOUGH_LINE_SHAPE_DETECTION_KERNEL_SOURCE"  _HEADER)
_embed_kernel("${_KERNELS_DIR}/shape-detection/HoughCircleShapeDetection.cl"
    "HOUGH_CIRCLE_SHAPE_DETECTION_KERNEL_SOURCE" _HEADER)

file(WRITE "${_GENERATED_DIR}/kernel_sources.h" "${_HEADER}")

add_library(config INTERFACE)

target_include_directories(config INTERFACE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/components
    ${_GENERATED_DIR}
)

target_compile_definitions(config INTERFACE
    ENABLE_LOGGING=$<BOOL:${ENABLE_LOGGING}>
)
